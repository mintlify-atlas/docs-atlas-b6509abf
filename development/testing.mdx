## Testing Overview

This guide covers testing strategies for the UCC Control de Acceso application. While the project currently doesn't have a comprehensive test suite, this document outlines best practices for implementing tests.

<Note>
  Testing infrastructure is planned for future implementation. This guide serves as a reference for when tests are added.
</Note>

## Testing Strategy

The application should follow a comprehensive testing approach:

```
┌─────────────────────────────────────┐
│     End-to-End Tests (E2E)         │  ← Test complete user flows
├─────────────────────────────────────┤
│     Integration Tests              │  ← Test module interactions
├─────────────────────────────────────┤
│     Unit Tests                     │  ← Test individual functions
└─────────────────────────────────────┘
```

## Unit Testing

### Testing Entities

Test domain entities for business logic:

```javascript src/modules/login/domain/entities/User.test.js
import { describe, it, expect } from 'vitest';
import { User } from './User';

describe('User Entity', () => {
  it('should create a user with correct properties', () => {
    const user = new User({
      id: 1,
      idInstitucional: '80123456',
      nombreCompleto: 'Juan Pérez',
      acceso: 'activo',
      totalFallas: 0,
      roles: ['estudiante']
    });

    expect(user.idInstitucional).toBe('80123456');
    expect(user.isBlocked()).toBe(false);
  });

  it('should identify blocked users', () => {
    const user = new User({
      id: 1,
      idInstitucional: '80123456',
      nombreCompleto: 'Juan Pérez',
      acceso: 'bloqueado',
      totalFallas: 4,
      roles: ['estudiante']
    });

    expect(user.isBlocked()).toBe(true);
  });

  it('should identify users at risk', () => {
    const user = new User({
      id: 1,
      idInstitucional: '80123456',
      nombreCompleto: 'Juan Pérez',
      acceso: 'activo',
      totalFallas: 3,
      roles: ['estudiante']
    });

    expect(user.isAtRisk()).toBe(true);
    expect(user.getRemainingAttempts()).toBe(1);
  });
});
```

### Testing Use Cases

```javascript src/modules/login/domain/usecases/ValidateInstitutionalIdUseCase.test.js
import { describe, it, expect, vi } from 'vitest';
import { ValidateInstitutionalIdUseCase } from './ValidateInstitutionalIdUseCase';

describe('ValidateInstitutionalIdUseCase', () => {
  it('should validate correct institutional ID', async () => {
    const mockRepo = {
      validateInstitutionalId: vi.fn().mockResolvedValue({
        id: 1,
        idInstitucional: '80123456',
        nombreCompleto: 'Juan Pérez',
        acceso: 'activo',
        totalFallas: 0,
        isBlocked: () => false
      })
    };

    const useCase = new ValidateInstitutionalIdUseCase(mockRepo);
    const result = await useCase.execute('80123456');

    expect(result.success).toBe(true);
    expect(result.user).toBeDefined();
  });

  it('should reject invalid format', async () => {
    const mockRepo = { validateInstitutionalId: vi.fn() };
    const useCase = new ValidateInstitutionalIdUseCase(mockRepo);
    
    const result = await useCase.execute('abc123');

    expect(result.success).toBe(false);
    expect(result.error).toContain('números');
  });

  it('should reject blocked users', async () => {
    const mockRepo = {
      validateInstitutionalId: vi.fn().mockResolvedValue({
        id: 1,
        idInstitucional: '80123456',
        acceso: 'bloqueado',
        isBlocked: () => true
      })
    };

    const useCase = new ValidateInstitutionalIdUseCase(mockRepo);
    const result = await useCase.execute('80123456');

    expect(result.success).toBe(false);
    expect(result.error).toContain('bloqueado');
  });
});
```

### Testing React Components

```javascript src/modules/login/presentation/components/LoginForm.test.jsx
import { describe, it, expect, vi } from 'vitest';
import { render, screen, fireEvent, waitFor } from '@testing-library/react';
import { LoginForm } from './LoginForm';

describe('LoginForm Component', () => {
  it('should render login form', () => {
    render(<LoginForm onSubmit={vi.fn()} />);
    
    expect(screen.getByLabelText(/ID Institucional/i)).toBeInTheDocument();
    expect(screen.getByRole('button', { name: /ingresar/i })).toBeInTheDocument();
  });

  it('should call onSubmit with institutional ID', async () => {
    const mockSubmit = vi.fn();
    render(<LoginForm onSubmit={mockSubmit} />);
    
    const input = screen.getByLabelText(/ID Institucional/i);
    const button = screen.getByRole('button', { name: /ingresar/i });

    fireEvent.change(input, { target: { value: '80123456' } });
    fireEvent.click(button);

    await waitFor(() => {
      expect(mockSubmit).toHaveBeenCalledWith('80123456');
    });
  });

  it('should show validation error for empty input', async () => {
    render(<LoginForm onSubmit={vi.fn()} />);
    
    const button = screen.getByRole('button', { name: /ingresar/i });
    fireEvent.click(button);

    await waitFor(() => {
      expect(screen.getByText(/requerido/i)).toBeInTheDocument();
    });
  });
});
```

### Testing Custom Hooks

```javascript src/modules/login/application/hooks/useLogin.test.js
import { describe, it, expect } from 'vitest';
import { renderHook, waitFor } from '@testing-library/react';
import { useLogin } from './useLogin';

describe('useLogin Hook', () => {
  it('should initialize with loading false', () => {
    const { result } = renderHook(() => useLogin());
    
    expect(result.current.loading).toBe(false);
    expect(result.current.error).toBeNull();
  });

  it('should handle successful login', async () => {
    const { result } = renderHook(() => useLogin());
    
    result.current.login('80123456');

    await waitFor(() => {
      expect(result.current.loading).toBe(false);
    });
  });
});
```

## E2E Testing

### Playwright Tests

```javascript tests/e2e/login-flow.spec.js
import { test, expect } from '@playwright/test';

test.describe('Login Flow', () => {
  test('should login successfully with valid ID', async ({ page }) => {
    await page.goto('http://localhost:3000/login');

    // Enter institutional ID
    await page.fill('input[name="idInstitucional"]', '80123456');
    await page.click('button[type="submit"]');

    // Should navigate to user profile
    await expect(page).toHaveURL(/\/user-profile/);
    await expect(page.locator('h1')).toContainText('Juan Pérez');
  });

  test('should show error for invalid ID', async ({ page }) => {
    await page.goto('http://localhost:3000/login');

    await page.fill('input[name="idInstitucional"]', '99999999');
    await page.click('button[type="submit"]');

    await expect(page.locator('.error-message')).toContainText('no encontrado');
  });

  test('should block access for blocked users', async ({ page }) => {
    await page.goto('http://localhost:3000/login');

    await page.fill('input[name="idInstitucional"]', '80999999');
    await page.click('button[type="submit"]');

    await expect(page.locator('.error-message')).toContainText('bloqueado');
  });
});
```

## Best Practices

<Tip>
  **Test Naming Convention**: Use descriptive names that explain what is being tested and the expected outcome.
  
  ```javascript
  // Good
  it('should reject login when user is blocked')
  
  // Bad
  it('test login')
  ```
</Tip>

<Warning>
  **Avoid Testing Implementation Details**: Focus on testing behavior, not internal implementation.
</Warning>

### Testing Principles

1. **Arrange-Act-Assert (AAA)** pattern
2. **Test one thing at a time**
3. **Use descriptive test names**
4. **Mock external dependencies**
5. **Keep tests isolated and independent**
6. **Avoid test interdependencies**

## Continuous Integration

Example GitHub Actions workflow:

```yaml .github/workflows/test.yml
name: Tests

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
          
      - name: Install dependencies
        run: npm ci
        
      - name: Run unit tests
        run: npm test
        
      - name: Run E2E tests
        run: npm run test:e2e
        
      - name: Generate coverage
        run: npm run test:coverage
```
